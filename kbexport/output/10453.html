-- fill in the begin date and end date to repost all the billable encounters
-- for that period

insert into p_patient_wp_item
(
patient_workplan_id,
cpr_id,
encounter_id,
workplan_id,
item_type,
ordered_service,
description,
ordered_by,
ordered_for,
owned_by,
status,
created_by
)
select 0,
cpr_id,
encounter_id,
0,
'Service',
'SENDBILLING',
'Resend Billing Data',
'#SYSTEM',
'#BILL',
'#BILL',
'DISPATCHED',
'#SYSTEM'
FROM p_patient_encounter
WHERE encounter_date > ''
and encounter_date <= ''
and bill_flag = 'Y'
and encounter_status = 'CLOSED'
GO

Notes

see also KB Article 10485 - Retry Posting in vs2 (This article will explain how to retry individual billing items)

 

The end date is not included in the time period, so if you want to repost all charges from Feb. 2, 2007 then the begin date is 02/02/2007 and the end date is 02/03/2007.
