Version 2:  following script retrievs and patients and their address from Centricity  for patients born before May 1,1995 and not given Adut / Child TD vaccine.

select distinct t.cpr_id,pt.billing_id,ge.patientid,first_name,last_name,convert(varchar(10),date_of_birth,101) as dob,ge.address1,ge.address2,ge.city,ge.state,ge.zip,ge.phone1 from c_procedure p
inner join p_treatment_item t
on t.procedure_id=p.procedure_id
inner join p_patient pt
on pt.cpr_id=t.cpr_id
left outer join paradigm..patientprofile ge
on ge.patientid=pt.billing_id
where p.cpt_code not in('90714','90715','90718')
and t.treatment_type='immunization'
and pt.date_of_birth < '5/1/1995'
